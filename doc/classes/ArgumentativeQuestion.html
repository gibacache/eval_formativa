def evaluate(params)
  ans = params[:answer].to_i == self.correct_answer
  arg = params[:argument].to_i == self.correct_argument
  correct = ans && arg
  feedback = ans ? self.feedback_wrong : self.feedback_wrong_argument if !correct
  return correct, feedback
end
